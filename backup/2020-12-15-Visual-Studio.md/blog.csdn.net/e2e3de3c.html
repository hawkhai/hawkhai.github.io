namespace "ATL" 没有成员 AtlGetCommCtrlVersion

hellokandy 2019-09-17 17:01:29 730 收藏 1
分类专栏： Windows 锦囊妙计 文章标签： AtlGetCommCtrlVersion

vs工程使用了wtl-80，提示 namespace "ATL" 没有成员 AtlGetCommCtrlVersion

解决方案：在atlapp.h文件里面，加入下面的代码

  1. // General DLL Version Helpers
  2. #pragma warning(push)
  3. #pragma warning(disable : 4191) // 'type cast' : unsafe conversion from 'FARPROC' to 'DLLGETVERSIONPROC'
  4. inline HRESULT AtlGetDllVersion(
  5. 	_In_ HINSTANCE hInstDLL,
  6. 	_Out_ DLLVERSIONINFO* pDllVersionInfo)
  7. {
  8. 	ATLENSURE(pDllVersionInfo != NULL);
  9. 	// We must get this function explicitly because some DLLs don't implement it.
  10. 	DLLGETVERSIONPROC pfnDllGetVersion = (DLLGETVERSIONPROC)::GetProcAddress(hInstDLL, "DllGetVersion");
  11. 	if (pfnDllGetVersion == NULL)
  12. 	{
  13. 		return E_NOTIMPL;
  14. 	}
  15. 	return (*pfnDllGetVersion)(pDllVersionInfo);
  16. }
  17. #pragma warning(pop)
  18. inline HRESULT AtlGetDllVersion(
  19. 	_In_z_ LPCTSTR lpstrDllName,
  20. 	_Out_ DLLVERSIONINFO* pDllVersionInfo)
  21. {
  22. 	HINSTANCE hInstDLL = ::LoadLibrary(lpstrDllName);
  23. 	if (hInstDLL == NULL)
  24. 	{
  25. 		return AtlHresultFromLastError();
  26. 	}
  27. 	HRESULT hRet = AtlGetDllVersion(hInstDLL, pDllVersionInfo);
  28. 	::FreeLibrary(hInstDLL);
  29. 	return hRet;
  30. }
  31. // Common Control Versions:
  32. //   WinNT 4.0          maj=4 min=00
  33. //   IE 3.x             maj=4 min=70
  34. //   IE 4.0             maj=4 min=71
  35. //   IE 5.0             maj=5 min=80
  36. //   Win2000            maj=5 min=81
  37. inline HRESULT AtlGetCommCtrlVersion(
  38. 	_Out_ LPDWORD pdwMajor,
  39. 	_Out_ LPDWORD pdwMinor)
  40. {
  41. 	ATLENSURE((pdwMajor != NULL) && (pdwMinor != NULL));
  42. 	DLLVERSIONINFO dvi;
  43. 	memset(&dvi, 0, sizeof(dvi));
  44. 	dvi.cbSize = sizeof(dvi);
  45. 	HRESULT hRet = AtlGetDllVersion(_T("comctl32.dll"), &dvi);
  46. 	if (SUCCEEDED(hRet))
  47. 	{
  48. 		*pdwMajor = dvi.dwMajorVersion;
  49. 		*pdwMinor = dvi.dwMinorVersion;
  50. 	}
  51. 	return hRet;
  52. }
  53. // Shell Versions:
  54. //   WinNT 4.0                                      maj=4 min=00
  55. //   IE 3.x, IE 4.0 without Web Integrated Desktop  maj=4 min=00
  56. //   IE 4.0 with Web Integrated Desktop             maj=4 min=71
  57. //   IE 4.01 with Web Integrated Desktop            maj=4 min=72
  58. //   Win2000                                        maj=5 min=00
  59. inline HRESULT AtlGetShellVersion(
  60. 	_Out_ LPDWORD pdwMajor,
  61. 	_Out_ LPDWORD pdwMinor)
  62. {
  63. 	ATLENSURE((pdwMajor != NULL) && (pdwMinor != NULL));
  64. 	DLLVERSIONINFO dvi;
  65. 	memset(&dvi, 0, sizeof(dvi));
  66. 	dvi.cbSize = sizeof(dvi);
  67. 	HRESULT hRet = AtlGetDllVersion(_T("shell32.dll"), &dvi);
  68. 	if (SUCCEEDED(hRet))
  69. 	{
  70. 		*pdwMajor = dvi.dwMajorVersion;
  71. 		*pdwMinor = dvi.dwMinorVersion;
  72. 	}
  73. 	return hRet;
  74. }

然后在使用的地方做相应修改

  1. inline bool IsCommCtrl6()
  2. {
  3. 	DWORD dwMajor = 0, dwMinor = 0;
  4. 	HRESULT hRet = /*ATL::*/AtlGetCommCtrlVersion(&dwMajor, &dwMinor);//Modify
  5. 	return (SUCCEEDED(hRet) && (dwMajor >= 6));
  6. }
