题号：5. 最长回文子串，难度中等

  1. class Solution {
  2. public:
  3.     string longestPalindrome(string s) {
  4.         int n = s.size();
  5.         vector<vector<int>> dp(n, vector<int>(n));
  6.         string ans;
  7.         for (int l = 0; l < n; ++l) {
  8.             for (int i = 0; i + l < n; ++i) {
  9.                 int j = i + l;
  10.                 if (l == 0) {
  11.                     dp[i][j] = 1;
  12.                 }
  13.                 else if (l == 1) {
  14.                     dp[i][j] = (s[i] == s[j]);
  15.                 }
  16.                 else {
  17.                     dp[i][j] = (s[i] == s[j] && dp[i + 1][j - 1]);
  18.                 }
  19.                 if (dp[i][j] && l + 1 > ans.size()) {
  20.                     ans = s.substr(i, l + 1);
  21.                 }
  22.             }
  23.         }
  24.         return ans;
  25.     }
  26. };
