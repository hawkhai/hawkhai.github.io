最近在做一些图片补全的工作。本以为补全就只有photoshop的patchmatch这一种方法，但后来发现不是这样。有一款号称linux下的ps的软件，名字叫做gimp。他也提供类似功能。细细一查，这款软件确实很有来头，连gtk在刚出现时都是专门为此软件写界面的。当然，现在gtk发展起来了。


这款软件很多功能都是靠插件在工作，他只提供一个通讯用的接口。靠着各种开源的插件，他实现了众多功能，好不好不敢说，不过和ps媲美是可以的。


好了，言归正传，resynthesizer就是一款用于补全的插件。网址在http://www.logarithmic.net/pfh/resynthesizer。大家可以去看看效果。作者也提供源代码，不过是插件的源代码。所以我修改了一下，使之能直接用在程序里。需要用opencv。


但遗憾的是，这个程序我并没有深入了解，不懂算法，只是会用而已。不想patchmatch那样，从算法到程序都搞懂了。希望有人能讲解一下其中的算法，不胜感激!

  11. //考虑3种情况，back， search， hole
  12. //back是不理会的区域
  13. //search是想要用之填补的区域
  14. //hole是要补的区域
  15. #define back 0
  16. #define search 100
  17. #define hole 255
  18.  
  19. //本程序的目的是修改gimp插件
  20. //证实gimp的像素索引从（0，0）开始
  21. //证实gimp的像素分布是rgb
  22.  
  23. typedef unsigned char Pixelel;
  24.  
  25. const int max_neighbours = 1000;
  26. const int max_trys_per_pixel = 10000;
  27.  
  28. /***********************/
  29. /***********************/
  30. /*** x,y Coordinates ***/
  31. /***********************/
  32. /***********************/
  33.  
  34. struct Coordinates {
  35. 	int x, y;
  36. 	Coordinates(int _x, int _y) :
  37. 			x(_x), y(_y) {
  38. 	}
  39.  
  40. 	Coordinates() {
  41. 	}
  42.  
  43. 	bool operator <(const Coordinates other) const {
  44. 		return y * y + x * x < (other.y * other.y + other.x * other.x);
  45. 	}
  46.  
  47. 	Coordinates operator +(const Coordinates a) const {
  48. 		return Coordinates(x + a.x, y + a.y);
  49. 	}
  50.  
  51. 	Coordinates operator -(const Coordinates a) const {
  52. 		return Coordinates(x - a.x, y - a.y);
  53. 	}
  54. };
  55.  
  56. /**************/
  57. /**************/
  58. /*** Bitmap ***/
  59. /**************/
  60. /**************/
  61.  
  62. //Bitmap class with three dimensions (width, height, number of channels)
  63. template<class t>
  64. struct Bitmap {
  65. 	int width, height, depth;
  66. 	t *data;
  67.  
  68. 	Bitmap() {
  69. 		data = 0;
  70. 	}
  71.  
  72. 	~Bitmap() {
  73. 		delete data;
  74. 	}
  75.  
  76. 	void size(int w, int h, int d) {
  77. 		width = w;
  78. 		height = h;
  79. 		depth = d;
  80.  
  81. 		delete data;
  82. 		data = new t[w * h * d];
  83. 	}
  84.  
  85. 	t *at(int x, int y) const {
  86. 		return data + (y * width + x) * depth;
  87. 	}
  88.  
  89. 	t *at(const Coordinates position) const {
  90. 		return at(position.x, position.y);
  91. 	}
  92.  
  93. 	//只会写回Dst
  94. 	void to_drawable(Mat drawable) {
  95. 		for (int i = 0; i < height; i++)
  96. 			for (int j = 0; j < width; j++)
  97. 				for (int k = 0; k < depth; k++)
  98. 					drawable.at<Vec3b>(i, j)[2 - k] = data[(i * width + j)
  99. 							* depth + k];
  100. 	}
  101.  
  102. 	//Src和Mask
  103. 	void from_drawable(Mat drawable) {
  104. 		if (drawable.channels() == 1) {
  105. 			for (int i = 0; i < height; i++)
  106. 				for (int j = 0; j < width; j++)
  107. 					data[i * width + j] = drawable.at<uchar>(i, j);
  108. 		} else {
  109. 			for (int i = 0; i < height; i++)
  110. 				for (int j = 0; j < width; j++)
  111. 					for (int k = 0; k < depth; k++)
  112. 						data[(i * width + j) * depth + k] = drawable.at<Vec3b>(
  113. 								i, j)[2 - k];
  114. 		}
  115. 	}
  116. };
  117.  
  118. /*******************************/
  119. /*******************************/
  120. /*** Status holding type *******/
  121. /*******************************/
  122. /*******************************/
  123.  
  124. struct Status {
  125. 	bool has_value, has_source;
  126. 	Coordinates source;
  127. };
  128.  
  129. /*******************************/
  130. /*******************************/
  131. /*** Get stuff from GIMP *******/
  132. /*******************************/
  133. /*******************************/
  134.  
  135. //Get a drawable and possibly its selection mask from the GIMP
  136. void fetch_image_and_mask(Mat drawable, Mat drawable_mask,
  137. 		Bitmap<Pixelel> &image, Bitmap<Pixelel> &mask) {
  138. 	assert(drawable.channels() == 3 &&drawable_mask.channels() == 1);
  139.  
  140. 	image.size(drawable.cols, drawable.rows, 3);
  141. 	mask.size(drawable.cols, drawable.rows, 1);
  142.  
  143. 	image.from_drawable(drawable);
  144. 	mask.from_drawable(drawable_mask);
  145. }
  146.  
  147. /******************/
  148. /******************/
  149. /*** Parameters ***/
  150. /******************/
  151. /******************/
  152.  
  153. struct Parameters {
  154. 	bool h_tile, v_tile;
  155. 	bool use_border;
  156.  
  157. 	double map_weight;
  158. 	double autism;
  159. 	int neighbours, trys;
  160. };
  161.  
  162. /* Restore the last parameters used from deep in the bowels of
  163.  the main GIMP app */
  164. void get_last_parameters(Parameters *param) {
  165. 	/* Defaults in case this is our first run */
  166. 	param->v_tile = true;
  167. 	param->h_tile = true;
  168. 	param->use_border = true;
  169. 	param->map_weight = 0.5;
  170. 	param->autism = 0.117; /* 30/256 */
  171. 	param->neighbours = 30;
  172. 	param->trys = 200;
  173. }
  174.  
  175. inline bool wrap_or_clip(Parameters ¶meters, Bitmap<Pixelel> &image,
  176. 		Coordinates &point) {
  177. 	while (point.x < 0)
  178. 		if (parameters.h_tile)
  179. 			point.x += image.width;
  180. 		else
  181. 			return false;
  182.  
  183. 	while (point.x >= image.width)
  184. 		if (parameters.h_tile)
  185. 			point.x -= image.width;
  186. 		else
  187. 			return false;
  188.  
  189. 	while (point.y < 0)
  190. 		if (parameters.v_tile)
  191. 			point.y += image.height;
  192. 		else
  193. 			return false;
  194.  
  195. 	while (point.y >= image.height)
  196. 		if (parameters.v_tile)
  197. 			point.y -= image.height;
  198. 		else
  199. 			return false;
  200.  
  201. 	return true;
  202. }
  203.  
  204. /* Helpers for the main function */
  205.  
  206. int diff_table[512], map_diff_table[512];
  207.  
  208. int input_bytes, map_bytes, map_pos, bytes;
  209. Bitmap<Pixelel> data, data_mask, corpus, corpus_mask;
  210. Bitmap<Status> status;
  211. vector<Coordinates> data_points, corpus_points, sorted_offsets;
  212.  
  213. Coordinates neighbours[max_neighbours];
  214. Pixelel neighbour_values[max_neighbours][8];
  215. Status *neighbour_statuses[max_neighbours];
  216. int n_neighbours;
  217.  
  218. int best;
  219. Coordinates best_point;
  220.  
  221. double neglog_cauchy(double x) {
  222. 	return log(x * x + 1.0);
  223. }
  224.  
  225. void make_offset_list(void) {
  226. 	int width = (corpus.width < data.width ? corpus.width : data.width);
  227. 	int height = (corpus.height < data.height ? corpus.height : data.height);
  228. 	sorted_offsets.resize(0);
  229. 	for (int y = -height + 1; y < height; y++)
  230. 		for (int x = -width + 1; x < width; x++)
  231. 			sorted_offsets.push_back(Coordinates(x, y));
  232.  
  233. 	sort(sorted_offsets.begin(), sorted_offsets.end());
  234. }
  235.  
  236. inline void try_point(const Coordinates &point) {
  237. 	int sum = 0;
  238.  
  239. 	for (int i = 0; i < n_neighbours; i++) {
  240. 		Coordinates off_point = point + neighbours[i];
  241. 		if (off_point.x < 0 || off_point.y < 0 || off_point.x >= corpus.width
  242. 				|| off_point.y >= corpus.height
  243. 				|| !corpus_mask.at(off_point)[0]) {
  244. 			sum += diff_table[0] * input_bytes + map_diff_table[0] * map_bytes;
  245. 		} else {
  246. 			const Pixelel *corpus_pixel = corpus.at(off_point), *data_pixel =
  247. 					neighbour_values[i];
  248. 			if (i)
  249. 				for (int j = 0; j < input_bytes; j++)
  250. 					sum += diff_table[256u + data_pixel[j] - corpus_pixel[j]];
  251. 			if (input_bytes != bytes)
  252. 				for (int j = input_bytes; j < bytes; j++)
  253. 					sum +=
  254. 							map_diff_table[256u + data_pixel[j]
  255. 									- corpus_pixel[j]];
  256. 		}
  257.  
  258. 		if (sum >= best)
  259. 			return;
  260. 	}
  261.  
  262. 	best = sum;
  263. 	best_point = point;
  264. }
  265.  
  266. /* This is the main function. */
  267.  
  268. Mat run(Mat src, Mat mask_search, Mat mask_hole) {
  269. 	assert(
  270. 			src.channels() == 3 &&mask_search.channels() == 1 &&mask_hole.channels() == 1);
  271. 	//search是search区域为0， 其他为255
  272. 	//hole是其他区域为0， hole为255
  273.  
  274. 	Parameters parameters;
  275.  
  276. 	Mat drawable = src.clone(), corpus_drawable = src.clone(), drawable_mask =
  277. 			mask_hole.clone(), corpus_drawable_mask = mask_search.clone();
  278.  
  279. 	srand(time(0));
  280.  
  281. 	get_last_parameters(¶meters);
  282.  
  283. 	input_bytes = 3;
  284. 	map_bytes = 0;
  285. 	map_pos = input_bytes;
  286. 	bytes = map_pos + map_bytes;
  287.  
  288. 	fetch_image_and_mask(drawable, drawable_mask, data, data_mask);
  289.  
  290. 	status.size(data.width, data.height, 1);
  291.  
  292. 	data_points.resize(0);
  293.  
  294. 	for (int y = 0; y < status.height; y++) {
  295. 		for (int x = 0; x < status.width; x++) {
  296. 			status.at(x, y)[0].has_source = false;
  297. 			status.at(x, y)[0].has_value = false;
  298.  
  299. 			if (parameters.use_border && data_mask.at(x, y)[0] == 0)
  300. 				status.at(x, y)->has_value = true;
  301.  
  302. 			if (data_mask.at(x, y)[0] != 0)
  303. 				data_points.push_back(Coordinates(x, y));
  304. 		}
  305. 	}
  306.  
  307. 	/* Fetch the corpus */
  308.  
  309. 	fetch_image_and_mask(corpus_drawable, corpus_drawable_mask, corpus,
  310. 			corpus_mask);
  311.  
  312. 	corpus_points.resize(0);
  313.  
  314. 	for (int y = 0; y < corpus.height; y++) {
  315. 		for (int x = 0; x < corpus.width; x++) {
  316. 			corpus_mask.at(x, y)[0] = 255 - corpus_mask.at(x, y)[0];
  317. 			if (corpus_mask.at(x, y)[0])
  318. 				corpus_points.push_back(Coordinates(x, y));
  319. 		}
  320. 	}
  321.  
  322. 	/* Sanity check */
  323.  
  324. 	if (!corpus_points.size() || !data_points.size()) {
  325. 		if (!corpus_points.size())
  326. 			cout << "The input texture is too small.";
  327. 		else
  328. 			cout << "The output image is too small.";
  329.  
  330. 		system("pause");
  331. 	}
  332.  
  333. 	/* Setup */
  334.  
  335. 	make_offset_list();
  336.  
  337. 	for (int i = -256; i < 256; i++) {
  338. 		double value = neglog_cauchy(i / 256.0 / parameters.autism)
  339. 				/ neglog_cauchy(1.0 / parameters.autism) * 65536.0;
  340. 		diff_table[256 + i] = int(value);
  341. 		map_diff_table[256 + i] = int(i * i * parameters.map_weight * 4.0);
  342. 	}
  343.  
  344. 	for (int i = 0; i < int(data_points.size()); i++) {
  345. 		int j = rand() % data_points.size();
  346. 		Coordinates temp = data_points[i];
  347. 		data_points[i] = data_points[j];
  348. 		data_points[j] = temp;
  349. 	}
  350.  
  351. 	for (int n = data_points.size(); n > 0;) {
  352. 		n = n * 3 / 4; // <- note magic number... the more repetition, the higher the quality, maybe
  353. 		for (int i = 0; i < n; i++)
  354. 			data_points.push_back(data_points[i]);
  355. 	}
  356.  
  357. 	/* Do it */
  358.  
  359. 	Bitmap<int> tried;
  360. 	tried.size(corpus.width, corpus.height, 1);
  361. 	for (int i = 0; i < corpus.width * corpus.height; i++)
  362. 		tried.data[i] = -1;
  363.  
  364. 	for (int i = data_points.size() - 1; i >= 0; i--) {
  365. 		if ((i & 1023) == 0)
  366. 			cout << (1.0 - float(i) / data_points.size()) << endl;
  367.  
  368. 		Coordinates position = data_points[i];
  369.  
  370. 		status.at(position)->has_value = true;
  371.  
  372. 		n_neighbours = 0;
  373. 		const int sorted_offsets_size = sorted_offsets.size();
  374. 		for (int j = 0; j < sorted_offsets_size; j++) {
  375. 			Coordinates point = position + sorted_offsets[j];
  376.  
  377. 			if (wrap_or_clip(parameters, data, point)
  378. 					&& status.at(point)->has_value) {
  379. 				neighbours[n_neighbours] = sorted_offsets[j];
  380. 				neighbour_statuses[n_neighbours] = status.at(point);
  381. 				for (int k = 0; k < bytes; k++)
  382. 					neighbour_values[n_neighbours][k] = data.at(point)[k];
  383. 				n_neighbours++;
  384. 				if (n_neighbours >= parameters.neighbours)
  385. 					break;
  386. 			}
  387. 		}
  388.  
  389. 		best = 1 << 30;
  390.  
  391. 		for (int j = 0; j < n_neighbours && best != 0; j++) {
  392. 			if (neighbour_statuses[j]->has_source) {
  393. 				Coordinates point = neighbour_statuses[j]->source
  394. 						- neighbours[j];
  395. 				if (point.x < 0 || point.y < 0 || point.x >= corpus.width
  396. 						|| point.y >= corpus.height)
  397. 					continue;
  398. 				if (!corpus_mask.at(point)[0] || tried.at(point)[0] == i)
  399. 					continue;
  400. 				try_point(point);
  401. 				tried.at(point)[0] = i;
  402. 			}
  403. 		}
  404.  
  405. 		for (int j = 0; j < parameters.trys && best != 0; j++)
  406. 			try_point(corpus_points[rand() % corpus_points.size()]);
  407.  
  408. 		for (int j = 0; j < input_bytes; j++)
  409. 			data.at(position)[j] = corpus.at(best_point)[j];
  410.  
  411. 		status.at(position)->has_source = true;
  412. 		status.at(position)->source = best_point;
  413. 	}
  414.  
  415. 	/* Write result to region */
  416. 	drawable.setTo(0);
  417. 	data.to_drawable(drawable);
  418.  
  419. 	return drawable;
  420. }
  421.  
  422. Mat RunImplement(Mat src, Mat mask) {
  423. 	assert(src.channels() == 3 &&mask.channels() == 1);
  424.  
  425. 	Mat mask_s = Mat(mask.size(), CV_8UC1, Scalar::all(255));
  426. 	Mat mask_h = Mat(mask.size(), CV_8UC1, Scalar::all(0));
  427.  
  428. 	for (int i = 0; i < mask.rows; i++) {
  429. 		for (int j = 0; j < mask.cols; j++) {
  430. 			if (mask.at<uchar>(i, j) == search)
  431. 				mask_s.at<uchar>(i, j) = 0;
  432.  
  433. 			if (mask.at<uchar>(i, j) == hole)
  434. 				mask_h.at<uchar>(i, j) = 255;
  435. 		}
  436. 	}
  437.  
  438. 	return run(src, mask_h, mask_s);
  439. }
