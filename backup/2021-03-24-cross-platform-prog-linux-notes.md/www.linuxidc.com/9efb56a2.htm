跨Linux和Windows的C编程杂记

[日期：2012-02-01] 来源：Linux社区  作者：naihe2010 [字体：大 中 小]

调用进程

　　Linux下可以直接使用system来调用外部程序。Windows最好使用WinExec，因为WinExec可以支持是打开还是隐藏程序窗口。用WinExec的第二个入参指明，如SW_SHOW/SW_HIDE。

杂项

　　Linux为srandom和random函数，Windows为srand和rand函数。

　　Linux为snprintf，Windows为_snprintf。

　　同理，Linux中的strcasecmp，Windows为_stricmp。

错误处理

　　Linux下面，通常使用全局变量errno来表示函数执行的错误号。Windows下要使用GetLastError ()调用来取得。

Linux环境下仅有的

　　这些函数或者宏，Windows中完全没有，需要用户手动实现。

　　atoll

[cpp] 
  1. long long  
  2. atoll (const char *p)  
  3. {  
  4.   int minus = 0;  
  5.   long long value = 0;  
  6.   if (*p == '-')  
  7.     {  
  8.       minus ++;  
  9.       p ++;  
  10.     }  
  11.   while (*p >= '0' && *p <= '9')  
  12.     {  
  13.       value *= 10;  
  14.       value += *p - '0';  
  15.       p ++;  
  16.     }  
  17.   return minus ? 0 - value : value;  
  18. }  

gettimeofday

[cpp] 
  1. #if defined(_MSC_VER) || defined(_MSC_EXTENSIONS)   
  2. #define EPOCHFILETIME  11644473600000000Ui64   
  3. #else   
  4. #define EPOCHFILETIME  11644473600000000ULL   
  5. #endif   
  6.   
  7. struct timezone  
  8. {  
  9.   int tz_minuteswest;  
  10.   int tz_dsttime;  
  11. };  
  12.   
  13. int  
  14. gettimeofday (struct timeval *tv, struct timezone *tz)  
  15. {  
  16.   FILETIME ft;  
  17.   LARGE_INTEGER li;  
  18.   __int64 t;  
  19.   static int tzflag;  
  20.   
  21.   if (tv)  
  22.     {  
  23.       GetSystemTimeAsFileTime (&ft);  
  24.       li.LowPart = ft.dwLowDateTime;  
  25.       li.HighPart = ft.dwHighDateTime;  
  26.       t = li.QuadPart;      /* In 100-nanosecond intervals */  
  27.       t -= EPOCHFILETIME;   /* Offset to the Epoch time */  
  28.       t /= 10;          /* In microseconds */  
  29.       tv->tv_sec = (long) (t / 1000000);  
  30.       tv->tv_usec = (long) (t % 1000000);  
  31.     }  
  32.   
  33.   if (tz)  
  34.     {  
  35.       if (!tzflag)  
  36.     {  
  37.       _tzset ();  
  38.       tzflag++;  
  39.     }  
  40.       tz->tz_minuteswest = _timezone / 60;  
  41.       tz->tz_dsttime = _daylight;  
  42.     }  
  43.   
  44.   return 0;  
  45. }
